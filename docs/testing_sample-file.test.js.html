 * Sample abstracted function to test sum between any two integers
 * @param {int} a 
 * @param {int} b

function testAdd(a, b){
	test(`${a} + ${b} = ${a+b}`, function(){
		expect(sum(a,b)).toBe(a+b);
	});
}

	for(let j = -4; j < 4; j++){
		testAdd(i, j);
	}
}
